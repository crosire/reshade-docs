197  template <typename T> inline T &get_private_data() const
198  {
199  uint64_t res;
200  get_private_data(reinterpret_cast<const uint8_t *>(&__uuidof(T)), &res);
201  return *reinterpret_cast<T *>(static_cast<uintptr_t>(res));
202  }
206  template <typename T> inline T &create_private_data()
207  {
208  uint64_t res = reinterpret_cast<uintptr_t>(new T());
209  set_private_data(reinterpret_cast<const uint8_t *>(&__uuidof(T)), res);
210  return *reinterpret_cast<T *>(static_cast<uintptr_t>(res));
211  }
215  template <typename T> inline void destroy_private_data()
216  {
217  uint64_t res;
218  get_private_data(reinterpret_cast<const uint8_t *>(&__uuidof(T)), &res);
219  delete reinterpret_cast<T *>(static_cast<uintptr_t>(res));
220  set_private_data(reinterpret_cast<const uint8_t *>(&__uuidof(T)), 0);
221  }
