25  inline HMODULE get_reshade_module_handle(HMODULE reshade_module = nullptr)
26  {
27  static HMODULE handle = reshade_module;
28  if (handle == nullptr)
29  {
30  HMODULE modules[1024]; DWORD num = 0;
31  if (K32EnumProcessModules(GetCurrentProcess(), modules, sizeof(modules), &num))
32  {
33  if (num > sizeof(modules))
34  num = sizeof(modules);
35 
36  for (DWORD i = 0; i < num / sizeof(HMODULE); ++i)
37  {
38  if (GetProcAddress(modules[i], "ReShadeRegisterAddon") &&
39  GetProcAddress(modules[i], "ReShadeUnregisterAddon"))
40  {
41  handle = modules[i];
42  break;
43  }
44  }
45  }
46  }
47  return handle;
48  }

165  inline bool register_addon(HMODULE addon_module, HMODULE reshade_module = nullptr)
166  {
167  addon_module = internal::get_current_module_handle(addon_module);
168  reshade_module = internal::get_reshade_module_handle(reshade_module);
169 
170  if (reshade_module == nullptr)
171  return false;
172 
173  // Check that the ReShade module supports the used API
174  const auto func = reinterpret_cast<bool(*)(HMODULE, uint32_t)>(
175  GetProcAddress(reshade_module, "ReShadeRegisterAddon"));
176  if (func == nullptr || !func(addon_module, RESHADE_API_VERSION))
177  return false;
178 
179 #if defined(IMGUI_VERSION_NUM)
180  // Check that the ReShade module was built with Dear ImGui support and supports the used version
181  const auto imgui_func = reinterpret_cast<const imgui_function_table *(*)(uint32_t)>(
182  GetProcAddress(reshade_module, "ReShadeGetImGuiFunctionTable"));
183  if (imgui_func == nullptr || !(imgui_function_table_instance() = imgui_func(IMGUI_VERSION_NUM)))
184  return false;
185 #endif
186 
187  return true;
188  }
