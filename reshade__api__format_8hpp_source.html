466  inline const uint32_t format_row_pitch(format value, uint32_t width)
467  {
468  if (value == format::unknown)
469  return 0;
470 
471  if (value <= format::r32g32b32a32_sint)
472  return 16 * width;
473  if (value <= format::r32g32b32_sint)
474  return 12 * width;
475  if (value <= format::x32_float_g8_uint)
476  return 8 * width;
477  if (value <= format::x24_unorm_g8_uint || value == format::l16a16_unorm)
478  return 4 * width;
479  if (value <= format::r16_sint || value == format::b5g6r5_unorm || value == format::b5g5r5a1_unorm || value == format::b5g5r5x1_unorm || value == format::l8a8_unorm || value == format::l16_unorm)
480  return 2 * width;
481  if (value <= format::a8_unorm || value == format::l8_unorm)
482  return 1 * width;
483  if (value <= format::g8r8_g8b8_unorm || (value >= format::b8g8r8a8_unorm && value <= format::b8g8r8x8_unorm_srgb) || (value == format::r8g8b8x8_unorm || value == format::r8g8b8x8_unorm_srgb))
484  return 4 * width;
485 
486  // Block compressed formats are bytes per block, rather than per pixel
487  if ((value >= format::bc1_typeless && value <= format::bc1_unorm_srgb) || (value >= format::bc4_typeless && value <= format::bc4_snorm))
488  return 8 * ((width + 3) / 4);
489  if ((value >= format::bc2_typeless && value <= format::bc2_unorm_srgb) || (value >= format::bc3_typeless && value <= format::bc3_unorm_srgb) || (value >= format::bc5_typeless && value <= format::bc7_unorm_srgb))
490  return 16 * ((width + 3) / 4);
491 
492  return 0;
493  }
